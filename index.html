                                          private static void intercala(int[] vetor, int inicio, int meio, int fim) {
                                            
                                            int novoVetor[] = new int[fim - inicio];
                                           
                                            int i = inicio;
                                    
                                            int m = meio;
                                            int pos = 0;
                                            while(i < meio && m < fim) {
                                              if(vetor[i] <= vetor[m]) {
                                                novoVetor[pos] = vetor[i];
                                                pos = pos + 1;
                                                i = i + 1;
                                              } else {
                                                novoVetor[pos] = vetor[m];
                                                pos = pos + 1;
                                                m = m + 1;
                                              }
                                            }
                                            while(i < meio) {
                                              novoVetor[pos] = vetor[i];
                                              pos = pos + 1;
                                              i = i + 1;
                                            }
                                            while(m < fim) {
                                              novoVetor[pos] = vetor[m];
                                              pos = pos + 1;
                                              m = m + 1;
                                            }
                                            for(pos = 0, i = inicio; i < fim; i++, pos++) {
                                              vetor[i] = novoVetor[pos];
                                            }
                                          }

                                                List[] buckets = new List[bktSize];

                                                }

                                                for(List bucket : buckets){  
                                                    for(int num : bucket)  {  
                                                        D[index++] = num;  
                                                    }  
                                                }

                                            }while(aux<3);

                                            public static int hash(int num, int bucketSize)  
                                            {  
                                            return num/bucketSize;  
                                            }
