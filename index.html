                                        public class Bubble {                                        
                                            public static void bubble(int vet[]) {
                                                for (int i = 0; i < vet.length; i++) {
                                                    boolean estaOrdenado = true;
                                                    for (int j = 0; j < vet.length - 1; j++) {
                                                        if (vet[j] > vet[j + 1]) {
                                                            int aux = vet[j];
                                                            vet[j] = vet[j + 1];
                                                            vet[j + 1] = aux;
                                                            estaOrdenado = false;
                                                        }
                                                    }
                                                    if (estaOrdenado) {
                                                        break;
                                                    }
                                                }
                                            }
